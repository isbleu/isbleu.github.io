💡 日周双周期协同策略机制

结合日线与周线级别的 factor_ema_momentum 因子：

  • 买入条件：日线 Target_Pos == 1 且 周线 Target_Pos == 1
  • 卖出条件：日线 Target_Pos == 0 且 周线 Target_Pos == 0
  • 维持持仓：出现 (1, 0) 或 (0, 1) 时保持前一日持仓状态不变

def calculate_weekly_factor(df_daily: pd.DataFrame, ema_period: int = 6) -> pd.Series:
    # 1. 按自然周聚合 OHLCV
    df_temp = df_daily.copy()
    df_temp['week_group'] = df_temp.index.to_period('W')
    weekly_last_dates = df_temp.reset_index().groupby('week_group')['date'].max()
    
    df_weekly = df_temp.groupby('week_group').agg({
        'open': 'first', 'high': 'max', 'low': 'min', 'close': 'last', 'volume': 'sum'
    })
    df_weekly.index = weekly_last_dates.values
    
    # 2. 计算周线 EMA 动量因子
    df_weekly_factored = factor_ema_momentum(df_weekly.copy(), ema_period=ema_period)
    
    # 3. reindex 到日线并 ffill 向下填充（周内沿用上一周已收盘周线持仓）
    weekly_pos_on_daily = df_weekly_factored['Target_Pos'].reindex(df_daily.index).ffill().fillna(0)
    return weekly_pos_on_daily
